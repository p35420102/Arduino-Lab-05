#include <Keypad.h>
#include "SevSeg.h"
SevSeg sevseg;
const byte rows = 4; /*¦æ*/
const byte cols = 4; /*¦C*/
char keys[rows][cols] = {
  {'F', 'E', 'D', 'C'},
  {'B', '3', '6', '9'},
  {'A', '2', '5', '8'},
  {'0', '1', '4', '7'}
};
byte rowPins[rows] = {11, 10, 9 , 8 };
byte colPins[cols] = {15, 14, 13, 12};
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, rows, cols); /*«Ø¥ßKeypad*/

void setup() {
  Serial.begin(9600);
  byte numDigits = 1;
  byte digitPins[] = {2};
  byte segmentPins[] = { 17, 16, 3, 4, 5, 6, 7} ;
  sevseg.begin(COMMON_CATHODE, numDigits, digitPins, segmentPins);
};
void loop() {
  char key = keypad.getKey();

  if (key != NO_KEY) {
    Serial.println(key);
    if ( key >= '0' && key <= '9') {
      sevseg.setNumber(key - '0', 1);
    }
  }
  sevseg.refreshDisplay();
};
